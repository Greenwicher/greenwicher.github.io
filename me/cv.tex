\cvitem{Title}{\emph{A Partition-based Random Search for Stochastic Multi-objective Optimization via Simulation}}
\cvitem{Supervisors}{A/Prof. Lee Loo Hay, A/Prof. Gao Siyang}
\cvitem{Published in}{Presented in INFORMS Annual Meetings, 2016}
\cvitem{Description}{
\begin{itemize}
    \item[--] Binary tree with nodes as partition rule is used to represent the strucutre of nested partitions.
    \item[--] Uniform sampling is implemented to collect information for each regions.
    \item[--] Most promising region is selected by comparing the promising index (average of domination count) of each region.
    \item[--] Partition the node of binary tree, which represents current most promising region. 
\end{itemize}}

\cvitem{Title}{\emph{Optimal Computing Budget Allocation to Select the Non-dominated Systems - a Large Deviations Perspective}}
\cvitem{Supervisors}{A/Prof. Lee Loo Hay, A/Prof Giullia Pedrielli}
\cvitem{Published in}{Under Review for IEEE Transactions on Automatic Control 2016}
\cvitem{Description}{
\begin{itemize}
    \item[--] Rate of probability of false selection considering multi-objective ranking \& selection problem is proposed. 
    \item[--] Optimal budget allocation strategy is derived by solving the non-convex optimization which maximizes the rate as a function of allocation proportion. 
    \item[--] Global optimal allocation strategy is proposed by decompoing the non-convex optimization into a set of convex optimization.
    \item[--] Numerical experiments show the newly proposed strategy (MOCBA+, MOCBA*) performs better than previous MOCBA. 
\end{itemize}}
